let templateList = []; //еӯҳеӮЁдёҙж—¶ж–Үд»¶еҲ—иЎЁ
this.fileList.forEach(async (item, index) => {
  if (!item.watermark) {
    // ејҖе§ӢеҠ ж°ҙеҚ°
    let base64 = await this.base64AddWaterMaker(newValue[index].content, this.waterMakeConfig);
    // жҠҠbase64иҪ¬дёәж–Үд»¶зұ»еһӢ з”ЁдәҺеҗҺз»ӯдёҠдј еҲ°еҗҺз«Ҝ
    let file = this.convertBase64UrlToBlob(base64);
    templateList.push({
      file,
      url: base64, // еӣҫзүҮзҡ„base64
      watermark: true //ж°ҙеҚ°ж ҮиҜҶ
    });
  } else {
    templateList.push(item);
    console.log("е·Із»Ҹж·»еҠ иҝҮж°ҙеҚ°");
  }
});
