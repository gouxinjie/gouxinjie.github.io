events жЁЎеқ—жҳҜ Node.js дёӯзҡ„дёҖдёӘж ёеҝғжЁЎеқ—пјҢз”ЁдәҺеӨ„зҗҶдәӢд»¶й©ұеҠЁзҡ„зј–зЁӢгҖӮе®ғдёәжҲ‘д»¬жҸҗдҫӣдәҶдёҖдёӘеҹәзЎҖзҡ„дәӢд»¶еҸ‘еёғ/и®ўйҳ…пјҲEventEmitterпјүжңәеҲ¶пјҢе…Ғи®ёеҜ№иұЎеҸ‘еҮәдәӢд»¶пјҢе№¶е…Ғи®ёе…¶д»–еҜ№иұЎзӣ‘еҗ¬е№¶е“Қеә”иҝҷдәӣдәӢд»¶гҖӮиҝҷз§ҚжңәеҲ¶е№ҝжіӣз”ЁдәҺ Node.js дёӯпјҢеҮ д№ҺжүҖжңүзҡ„ж ёеҝғжЁЎеқ—йғҪдҪҝз”ЁдәҶдәӢд»¶й©ұеҠЁжЁЎеһӢгҖӮ

2. EventEmitter зұ» вҖӢ

EventEmitter жҳҜ events жЁЎеқ—дёӯзҡ„ж ёеҝғзұ»пјҢз”ЁдәҺеҲӣе»әдәӢд»¶еҸ‘е°„еҷЁеҜ№иұЎгҖӮйҖҡиҝҮ EventEmitter зұ»пјҢдҪ еҸҜд»Ҙпјҡ

  • и§ҰеҸ‘дәӢд»¶пјҲemitпјүгҖӮ
  • зӣ‘еҗ¬дәӢд»¶пјҲonпјүгҖӮ
  • з§»йҷӨдәӢд»¶зӣ‘еҗ¬еҷЁпјҲoff жҲ– removeListenerпјүгҖӮ

3. еёёз”Ёж–№жі• вҖӢ

3.1 emitter.on(eventName, listener) вҖӢ

on() ж–№жі•з”ЁдәҺдёәжҢҮе®ҡдәӢд»¶жіЁеҶҢдёҖдёӘзӣ‘еҗ¬еҷЁгҖӮеҪ“дәӢд»¶иў«и§ҰеҸ‘ж—¶пјҢзӣ‘еҗ¬еҷЁдјҡжү§иЎҢгҖӮ

javascript
const EventEmitter = require("events");
const emitter = new EventEmitter();

// жіЁеҶҢдәӢд»¶зӣ‘еҗ¬еҷЁ
emitter.on("greet", (name) => {
  console.log(`Hello, ${name}!`);
});

// и§ҰеҸ‘дәӢд»¶
emitter.emit("greet", "Alice"); // иҫ“еҮәпјҡHello, Alice!

3.2 emitter.emit(eventName, ...args) вҖӢ

emit() ж–№жі•з”ЁдәҺи§ҰеҸ‘жҢҮе®ҡзҡ„дәӢд»¶пјҢе№¶дј йҖ’еҸӮж•°з»ҷдәӢд»¶еӨ„зҗҶеҮҪж•°гҖӮдёҖдёӘдәӢд»¶еҸҜд»Ҙиў«и§ҰеҸ‘еӨҡж¬ЎгҖӮ

const emitter = new EventEmitter();

emitter.on("greet", (name) => {
  console.log(`Hello, ${name}!`);
});

emitter.emit("greet", "Bob"); // иҫ“еҮәпјҡHello, Bob!

3.3 emitter.once(eventName, listener) вҖӢ

once() ж–№жі•дёҺ on() зұ»дјјпјҢдёҚеҗҢд№ӢеӨ„еңЁдәҺпјҢonce() жіЁеҶҢзҡ„дәӢд»¶зӣ‘еҗ¬еҷЁеңЁдәӢд»¶иў«и§ҰеҸ‘еҗҺпјҢеҸӘдјҡжү§иЎҢдёҖж¬ЎпјҢд№ӢеҗҺдјҡиҮӘеҠЁз§»йҷӨгҖӮ

javascript
const emitter = new EventEmitter();

emitter.once("greet", (name) => {
  console.log(`Hello, ${name}!`);
});

emitter.emit("greet", "David"); // иҫ“еҮәпјҡHello, David!

3.4 emitter.removeListener(eventName, listener) вҖӢ

removeListener() ж–№жі•з”ЁдәҺз§»йҷӨжҢҮе®ҡдәӢд»¶зҡ„зӣ‘еҗ¬еҷЁгҖӮд№ҹеҸҜд»ҘдҪҝз”Ё off() ж–№жі•пјҢе®ғжҳҜ removeListener() зҡ„еҲ«еҗҚгҖӮ

javascript
const emitter = new EventEmitter();

const greetListener = (name) => {
  console.log(`Hello, ${name}!`);
};

emitter.on("greet", greetListener);
emitter.emit("greet", "Frank"); // иҫ“еҮәпјҡHello, Frank!
emitter.removeListener("greet", greetListener);
emitter.emit("greet", "Grace"); // д»Җд№ҲйғҪдёҚиҫ“еҮә

3.5 emitter.removeAllListeners([eventName]) вҖӢ

removeAllListeners() з”ЁдәҺз§»йҷӨжҢҮе®ҡдәӢд»¶зҡ„жүҖжңүзӣ‘еҗ¬еҷЁгҖӮеҰӮжһңжІЎжңүжҸҗдҫӣдәӢд»¶еҗҚпјҢеҲҷз§»йҷӨжүҖжңүдәӢд»¶зҡ„жүҖжңүзӣ‘еҗ¬еҷЁгҖӮ

6. ејӮеёёдәӢд»¶пјҡerror вҖӢ

EventEmitter е®һдҫӢиҝҳдјҡи§ҰеҸ‘ error дәӢд»¶гҖӮеҰӮжһң error дәӢд»¶жІЎжңүиў«зӣ‘еҗ¬пјҢе№¶дё”и§ҰеҸ‘дәҶиҜҘдәӢд»¶пјҢзЁӢеәҸдјҡжҠӣеҮәејӮеёёе№¶йҖҖеҮәгҖӮдёәдәҶйҒҝе…ҚзЁӢеәҸеҙ©жәғпјҢдҪ еҸҜд»ҘжҚ•иҺ· error дәӢд»¶пјҡ

javascript
const emitter = new EventEmitter();

// жіЁеҶҢ 'error' дәӢд»¶
emitter.on("error", (err) => {
  console.log("жҚ•иҺ·еҲ°й”ҷиҜҜ:", err);
});

// и§ҰеҸ‘ 'error' дәӢд»¶
emitter.emit("error", new Error("Something went wrong!"));
