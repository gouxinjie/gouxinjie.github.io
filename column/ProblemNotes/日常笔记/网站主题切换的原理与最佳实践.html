3. JavaScript еҲҮжҚўйҖ»иҫ‘ вҖӢ

е®һзҺ°дё»йўҳеҲҮжҚўзҡ„ж ёеҝғ JavaScript д»Јз Ғпјҡ

javascript
class ThemeManager {
  constructor() {
    this.body = document.body;
    this.themeButtons = document.querySelectorAll(".theme-btn");
    this.init();
  }

  init() {
    // еҠ иҪҪдҝқеӯҳзҡ„дё»йўҳ
    this.loadTheme();
    // з»‘е®ҡдәӢд»¶
    this.bindEvents();
    // зӣ‘еҗ¬зі»з»ҹдё»йўҳеҸҳеҢ–
    this.watchSystemTheme();
  }

  bindEvents() {
    this.themeButtons.forEach((button) => {
      button.addEventListener("click", (e) => {
        const theme = e.target.dataset.theme;
        this.setTheme(theme);
      });
    });
  }

  setTheme(theme) {
    // з§»йҷӨжүҖжңүдё»йўҳзұ»
    this.body.classList.remove("theme-light", "theme-dark");
    // ж·»еҠ ж–°дё»йўҳзұ»
    this.body.classList.add(theme);

    // жӣҙж–°жҢүй’®зҠ¶жҖҒ
    this.updateButtonStates(theme);

    // дҝқеӯҳеҲ°жң¬ең°еӯҳеӮЁ
    this.saveTheme(theme);
  }

  updateButtonStates(activeTheme) {
    this.themeButtons.forEach((button) => {
      button.classList.toggle("active", button.dataset.theme === activeTheme);
    });
  }

  saveTheme(theme) {
    localStorage.setItem("user-theme", theme);
  }

  loadTheme() {
    const savedTheme = localStorage.getItem("user-theme");
    if (savedTheme) {
      this.setTheme(savedTheme);
    }
  }

  watchSystemTheme() {
    if (window.matchMedia) {
      const mediaQuery = window.matchMedia("(prefers-color-scheme: dark)");

      mediaQuery.addEventListener("change", (e) => {
        // еҸӘжңүз”ЁжҲ·жІЎжңүжүӢеҠЁйҖүжӢ©ж—¶жүҚи·ҹйҡҸзі»з»ҹ
        if (!localStorage.getItem("user-theme")) {
          this.setTheme(e.matches ? "theme-dark" : "theme-light");
        }
      });
    }
  }
}

// еҲқе§ӢеҢ–дё»йўҳз®ЎзҗҶеҷЁ
new ThemeManager();

            // дҝқеӯҳз”ЁжҲ·йҖүжӢ©еҲ°localStorage
            localStorage.setItem("theme", theme);
          });
        });

        function setTheme(theme) {
          console.log("theme:", theme);

          // з§»йҷӨжүҖжңүдё»йўҳзұ»
          body.classList.remove("theme-light", "theme-dark", "theme-blue");
          // ж·»еҠ еҪ“еүҚдё»йўҳзұ»
          body.classList.add(theme);

          // жӣҙж–°жҢүй’®зҡ„activeзҠ¶жҖҒ
          themeButtons.forEach((button) => {
            if (button.getAttribute("data-theme") === theme) {
              button.classList.add("active");
            } else {
              button.classList.remove("active");
            }
          });
        }

        // зӣ‘еҗ¬зі»з»ҹдё»йўҳеҸҳеҢ–пјҲеҸҜйҖүеҠҹиғҪпјү
        if (window.matchMedia) {

          darkModeMediaQuery.addEventListener("change", (e) => {
            // еҸӘжңүеҪ“з”ЁжҲ·жІЎжңүжүӢеҠЁйҖүжӢ©дё»йўҳж—¶пјҢжүҚи·ҹйҡҸзі»з»ҹ
            if (!localStorage.getItem("theme")) {
              if (e.matches) {
                setTheme("theme-dark");
              } else {
                setTheme("theme-light");
              }
            }
          });
        }
      });
